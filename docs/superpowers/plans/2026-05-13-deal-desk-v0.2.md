```typescript
// DEAL DESK: Phase 6 v0.2 — edit thesis
const updateThesisInputSchema = z.object({
  name: z.string().min(1).max(255).optional(),
  sector: z.string().min(1).max(255).optional(),
  subSectors: z.array(z.string()).optional(),
  geos: z.array(z.string()).optional(),
  revenueMin: z.number().optional().nullable(),
  revenueMax: z.number().optional().nullable(),
  ebitdaMin: z.number().optional().nullable(),
  ebitdaMax: z.number().optional().nullable(),
  dealSizeMin: z.number().optional().nullable(),
  dealSizeMax: z.number().optional().nullable(),
  ownershipPreferences: z.array(z.string()).optional(),
  exclusionCriteria: z.string().optional().nullable(),
  narrative: z.string().optional().nullable(),
  status: z.enum(["active", "paused", "archived"]).optional(),
});

router.patch("/:companyId/deal-desk/theses/:thesisId", async (req, res) => {
  const companyId = req.params.companyId;
  const thesisId = req.params.thesisId;
  if (!companyId || !thesisId) {
    res.status(400).json({ error: "missing params" });
    return;
  }
  await assertCompanyAccess(req, companyId);
  const input = updateThesisInputSchema.parse(req.body);

  const updateValues: Record<string, unknown> = { ...input, updatedAt: new Date() };
  const numericFields = ["revenueMin", "revenueMax", "ebitdaMin", "ebitdaMax", "dealSizeMin", "dealSizeMax"];
  for (const field of numericFields) {
    if (updateValues[field] !== undefined && updateValues[field] !== null) {
      updateValues[field] = String(updateValues[field]);  // Drizzle numeric expects string
    }
  }

  const [updated] = await db
    .update(ddTheses)
    .set(updateValues)
    .where(and(eq(ddTheses.id, thesisId), eq(ddTheses.paperclipCompanyId, companyId)))
    .returning();

  if (!updated) {
    res.status(404).json({ error: "thesis not found" });
    return;
  }
  res.json({ thesis: updated });
});
```

```tsx
// DEAL DESK: Phase 6 v0.2 — shared thesis form used by onboarding and edit dialog
import { useState, useEffect } from "react";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { Textarea } from "@/components/ui/textarea";
import { Button } from "@/components/ui/button";

export interface ThesisFormValues {
  name: string;
  sector: string;
  geos: string;  // comma-separated; serialize/deserialize at the boundary
  revenueMin: string;
  revenueMax: string;
  ownershipPreferences: string[];
  narrative: string;
  templateSlug: string | null;
}

export const EMPTY_THESIS: ThesisFormValues = {
  name: "",
  sector: "",
  geos: "",
  revenueMin: "",
  revenueMax: "",
  ownershipPreferences: [],
  narrative: "",
  templateSlug: null,
};

interface Props {
  initial?: ThesisFormValues;
  onSubmit: (values: ThesisFormValues) => Promise<void> | void;
  submitting?: boolean;
  submitLabel?: string;
  showTemplates?: boolean;
}

export function ThesisForm({
  initial,
  onSubmit,
  submitting = false,
  submitLabel = "Save thesis",
  showTemplates = true,
}: Props) {
  const [values, setValues] = useState<ThesisFormValues>(initial ?? EMPTY_THESIS);
  useEffect(() => { if (initial) setValues(initial); }, [initial]);

  function update<K extends keyof ThesisFormValues>(key: K, val: ThesisFormValues[K]) {
    setValues((prev) => ({ ...prev, [key]: val }));
  }

  return (
    <form
      className="space-y-4"
      onSubmit={(e) => { e.preventDefault(); void onSubmit(values); }}
    >
      <div>
        <Label htmlFor="thesis-name">Thesis name</Label>
        <Input
          id="thesis-name"
          value={values.name}
          onChange={(e) => update("name", e.target.value)}
          placeholder={values.sector ? `${values.sector} thesis` : "e.g., HVAC Southeast"}
        />
      </div>

      <div>
        <Label htmlFor="thesis-sector">Sector *</Label>
        <Input
          id="thesis-sector"
          value={values.sector}
          onChange={(e) => update("sector", e.target.value)}
          required
        />
      </div>

      <div>
        <Label htmlFor="thesis-geos">Geographies (comma-separated)</Label>
        <Textarea
          id="thesis-geos"
          rows={2}
          value={values.geos}
          onChange={(e) => update("geos", e.target.value)}
          placeholder="FL, GA, NC, SC, TN, AL"
        />
      </div>

      <div className="grid grid-cols-2 gap-3">
        <div>
          <Label htmlFor="thesis-rev-min">Revenue min ($)</Label>
          <Input
            id="thesis-rev-min"
            inputMode="numeric"
            value={values.revenueMin}
            onChange={(e) => update("revenueMin", e.target.value)}
          />
        </div>
        <div>
          <Label htmlFor="thesis-rev-max">Revenue max ($)</Label>
          <Input
            id="thesis-rev-max"
            inputMode="numeric"
            value={values.revenueMax}
            onChange={(e) => update("revenueMax", e.target.value)}
          />
        </div>
      </div>

      <div>
        <Label>Ownership preferences</Label>
        <div className="flex flex-wrap gap-3 text-sm">
          {(["Founder-owned", "Family-owned", "Sponsor-backed"] as const).map((opt) => (
            <label key={opt} className="flex items-center gap-2">
              <input
                type="checkbox"
                checked={values.ownershipPreferences.includes(opt)}
                onChange={(e) => {
                  const next = e.target.checked
                    ? [...values.ownershipPreferences, opt]
                    : values.ownershipPreferences.filter((x) => x !== opt);
                  update("ownershipPreferences", next);
                }}
              />
              {opt}
            </label>
          ))}
        </div>
      </div>

      <div>
        <Label htmlFor="thesis-narrative">Narrative</Label>
        <Textarea
          id="thesis-narrative"
          rows={4}
          value={values.narrative}
          onChange={(e) => update("narrative", e.target.value)}
          placeholder="Describe your investment thesis in your own words."
        />
      </div>

      <Button type="submit" disabled={submitting || !values.sector}>
        {submitting ? "Saving…" : submitLabel}
      </Button>
    </form>
  );
}

export function thesisFormValuesToApiPayload(v: ThesisFormValues) {
  return {
    name: v.name || `${v.sector} thesis`,
    sector: v.sector,
    geos: v.geos.split(",").map((g) => g.trim()).filter(Boolean),
    revenueMin: v.revenueMin ? Number(v.revenueMin) : null,
    revenueMax: v.revenueMax ? Number(v.revenueMax) : null,
    ownershipPreferences: v.ownershipPreferences,
    narrative: v.narrative || null,
    templateSlug: v.templateSlug,
  };
}
```

```tsx
// DEAL DESK: Phase 6 v0.2 — modal wrapper around ThesisForm
import { Dialog, DialogContent, DialogHeader, DialogTitle } from "@/components/ui/dialog";
import { useMutation, useQueryClient } from "@tanstack/react-query";
import { dealDeskApi } from "@/api/dealDesk";
import { queryKeys } from "@/lib/queryKeys";
import type { DdThesis } from "@/api/dealDesk";
import {
  ThesisForm,
  thesisFormValuesToApiPayload,
  type ThesisFormValues,
} from "./ThesisForm";

interface Props {
  open: boolean;
  onOpenChange: (open: boolean) => void;
  companyId: string;
  thesis: DdThesis;
}

export function EditThesisDialog({ open, onOpenChange, companyId, thesis }: Props) {
  const queryClient = useQueryClient();

  const initial: ThesisFormValues = {
    name: thesis.name,
    sector: thesis.sector,
    geos: thesis.geos.join(", "),
    revenueMin: thesis.revenueMin?.toString() ?? "",
    revenueMax: thesis.revenueMax?.toString() ?? "",
    ownershipPreferences: thesis.ownershipPreferences,
    narrative: thesis.narrative ?? "",
    templateSlug: thesis.templateSlug ?? null,
  };

  const mutation = useMutation({
    mutationFn: (values: ThesisFormValues) =>
      dealDeskApi.updateThesis(companyId, thesis.id, thesisFormValuesToApiPayload(values)),
    onSuccess: () => {
      void queryClient.invalidateQueries({ queryKey: queryKeys.dealDesk.thesis(companyId, thesis.id) });
      void queryClient.invalidateQueries({ queryKey: queryKeys.dealDesk.theses(companyId) });
      onOpenChange(false);
    },
  });

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="max-w-xl">
        <DialogHeader>
          <DialogTitle>Edit thesis</DialogTitle>
        </DialogHeader>
        <ThesisForm
          initial={initial}
          onSubmit={(values) => mutation.mutateAsync(values).then(() => undefined)}
          submitting={mutation.isPending}
          submitLabel="Save changes"
          showTemplates={false}
        />
      </DialogContent>
    </Dialog>
  );
}
```
